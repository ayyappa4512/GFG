<pre style="user-select: auto;"><strong style="user-select: auto;"><span style="font-size: 18px; user-select: auto;">Input: </span></strong><span style="font-size: 18px; user-select: auto;">a[] = [1, 2, 4, 3, 5, 6], b[] = [3, 4, 5, 6, 7]
